
query queryNeedRelayBlockPage {
  justificationMappings (
    first: 5
    orderBy: BLOCK_NUMBER_DESC
  ) {
    nodes {
      id
      blockNumber
      blockHash
      mandatory
      justification
    }
  }
  needRelayBlocks (
    first: 5
    orderBy: BLOCK_NUMBER_ASC
  ) {
    nodes {
      id
      blockNumber
      blockHash
      type
      origin
      laneId
      messageNonce
      timestamp
    }
  }
}
